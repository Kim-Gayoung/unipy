	PORT = 8888

		while True:
			print ("=====[1] Take Photo, [2] Start/Stop Streaming, [3] Turn Left(Camera), [4] Turn Right(Camera)=====")
			select = input("Select Command : ")
			
			if select == '1':
				# take photo(custom) & save photo in cloud
				pass
			elif select == '2':
				# streaming
				pass
			elif select == '3':
				# turn left
				sendControlMessage('l')
			elif select == '4':

				sendControlMessage('r')
			else:
				break
